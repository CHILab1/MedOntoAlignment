SELECT DISTINCT ?LocationCategory 
WHERE {
  ?LocationCategory rdfs:subClassOf ?mid1 .
  ?mid1 rdfs:subClassOf ?mid2 .
  ?mid2 rdfs:subClassOf icd:1154280071 .

  icd:NameLocation a ?class .
  ?class rdfs:subClassOf+ ?LocationCategory .
  
}

Where: 
1154280071 = Anatomy and topography
NameLocation = choose the ICD-11 Code of a specific location
